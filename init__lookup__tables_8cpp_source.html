510  for (auto i = 0; i < id_q1; i++)
511  rndq1_start += quarks[i].number_of_rnd_vec;
512  auto rndq2_start = 0;
513  for (auto i = 0; i < id_q2; i++)
514  rndq2_start += quarks[i].number_of_rnd_vec;
515 
516  auto rndq1_end = rndq1_start + quarks[id_q1].number_of_rnd_vec;
517  auto rndq2_end = rndq2_start + quarks[id_q2].number_of_rnd_vec;
518 
519  // check if there are enough random vectors

521  (id_q1 == id_q2 && quarks[id_q1].number_of_rnd_vec < 2)) {
522  std::cerr << "There are not enough random vectors for charged correlators"
523  << std::endl;
524  exit(-1);
525  }
526 
527  // finally filling the array
528  std::vector<std::pair<ssize_t, ssize_t>> rnd_vec_comb;
529  if (!C1) {
530  for (ssize_t i = rndq1_start; i < rndq1_end; ++i)
531  for (ssize_t j = rndq2_start; j < rndq2_end; ++j)
532  // To avoid bias, different random vectors must have different indices.
533  if (i != j) {
534  rnd_vec_comb.emplace_back(i, j);

539  // combinations are possible
540  rnd_vec_comb.emplace_back(i, i);

542 
543  return rnd_vec_comb;
