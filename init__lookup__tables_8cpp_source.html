42  Correlators_2 const &correlator,
43  Operator_list const &operator_list,
44  std::vector<std::vector<QuantumNumbers>> &quantum_numbers,
45  std::map<int, int> const &momentum_cutoff) {
46  std::vector<Operators> qn_op;
47  for (auto const &op_number : correlator.operator_numbers) {
48  if (op_number >= ssize(operator_list)) {
49  std::ostringstream oss;
50  oss << "Operator with ID " << op_number
51  << " which is used in [correlator_lists] is not defined. Please adjust your "
52  "parameter file.";
53  throw std::runtime_error(oss.str());
54  }
55  qn_op.emplace_back(operator_list[op_number]);
56  }
57 
58  std::cout << "Constructing momentum combinations for " << correlator.type << std::endl;
59 
60  // The following data structure holds the source and sink vertex indices in
61  // the diagrams. The first part of the pair are the source vertices, the
62  // second part are the sink vertices.
63  using Vertices = std::pair<std::vector<int>, std::vector<int>>;
64  std::map<std::string, Vertices> diagram_vertices;
65 
66  diagram_vertices["C1"] = Vertices({0}, {});
67  diagram_vertices["C20"] = Vertices({0}, {1});
68  diagram_vertices["C20V"] = Vertices({0}, {1});
69  diagram_vertices["C2c"] = Vertices({0}, {1});
70  diagram_vertices["C30"] = Vertices({0, 2}, {1});
71  diagram_vertices["C30V"] = Vertices({0, 1}, {2});
72  diagram_vertices["C3c"] = Vertices({0, 2}, {1});
73  diagram_vertices["C40B"] = Vertices({0, 3}, {1, 2});
74  diagram_vertices["C40C"] = Vertices({0, 2}, {1, 3});
75  diagram_vertices["C40D"] = Vertices({0, 2}, {1, 3});
76  diagram_vertices["C40V"] = Vertices({0, 1}, {2, 3});
77  diagram_vertices["C4cB"] = Vertices({0, 3}, {1, 2});
78  diagram_vertices["C4cC"] = Vertices({0, 2}, {1, 3});
79  diagram_vertices["C4cD"] = Vertices({0, 2}, {1, 3});
80  diagram_vertices["C4cV"] = Vertices({0, 1}, {2, 3});
81  diagram_vertices["Check"] = Vertices({0}, {1});
82 
83  // We extract the vertices from the given diagram. As this is not critical in
84  // performance we treat us with boundary checks here via `at` instead of
85  // `operator[]`.
86  auto const &vertices = diagram_vertices.at(correlator.type);
87 
88  // The contractions that have to be done are all combinations of the various
89  // possibilities from each vertex. We filter by momentum conservation later.
90  // For the cartesian product of all operator sets we first need the sequence
91  // of operators at each vertex.
92  std::vector<ssize_t> lengths_source(vertices.first.size());
93  std::transform(vertices.first.begin(),
94  vertices.first.end(),
95  lengths_source.begin(),
96  [&qn_op](int const i) { return ssize(qn_op.at(i)); });
97  CartesianProduct product_source(lengths_source);
98 
99  // Same for the sink vertices.
100  std::vector<ssize_t> lengths_sink(vertices.first.size());
101  std::transform(vertices.first.begin(),
102  vertices.first.end(),
103  lengths_sink.begin(),
104  [&qn_op](int const i) { return ssize(qn_op.at(i)); });
105  CartesianProduct product_sink(lengths_sink);
106 
107  // Space to store the quantum numbers assembled from the given index
108  // combination. These are defined here to avoid many allocations.
109  std::vector<QuantumNumbers> qn_source(vertices.first.size());
110  std::vector<QuantumNumbers> qn_sink(vertices.second.size());
111  std::vector<QuantumNumbers> qn_all(qn_source.size() + qn_sink.size());
112 
113  // We iterate over all source index combinations.
114  for (auto const &indices_source : product_source) {
115  // From the given index set we need to extract the operators. This is a bit
116  // tricky as there are two levels of indexing here. The index `j` just
117  // counts the source vertices of the diagram, always starting from 0.
118  for (int j = 0; j != ssize(vertices.first); ++j) {
119  // Then we need to get the vertex index (spanning source and sink) via
120  // the data structure. This lets us retrieve the set of operators for
121  // that vertex.
122  auto const &ops = qn_op.at(vertices.first.at(j));
123 
124  // From that operator we take the one that is specified in the current
125  // index set from the cartesian product.
126  qn_source[j] = ops.at(indices_source[j]);
127 
128  // Perhaps slightly inconsistent the operators are not grouped by source
129  // and sink in the assembly code much later. Therefore we need to copy
130  // the operator into this second temporary data structure where it is
131  // indexed with the global vertex index.
132  qn_all[vertices.first.at(j)] = qn_source[j];
133  }
134 
135  // Compute the total source momentum.
136  Vector p_so{0, 0, 0};
137  int sum_norm_sq = 0;
138  for (auto const &q : qn_source) {
139  p_so += q.momentum;
140  sum_norm_sq += q.momentum.squaredNorm();
141  }
142 
143  // If the operator was specified with a total momentum selection but the
144  // current one is not in that list, discard the current combination of
145  // operators.
146  if (!correlator.tot_mom.empty() &&
147  std::find(correlator.tot_mom.begin(), correlator.tot_mom.end(), p_so) ==
148  correlator.tot_mom.end()) {
149  continue;
150  }
151 
152  // Also discard when we are beyond the momentum cutoff.
153  if (sum_norm_sq > momentum_cutoff.at(p_so.squaredNorm())) {
154  continue;
155  }
156 
157  // The `C1` diagram is special as it is source-only. We do not need to work
158  // with sink operators at all, then. Enforcing momentum conservation is not
159  // sensible in this context.
160  if (vertices.second.empty()) {
161  quantum_numbers.push_back(qn_source);
162  } else {
163  // Analogously iterate through the sink vertex operators.
164  for (auto const &indices_sink : product_sink) {
165  for (int j = 0; j != ssize(vertices.second); ++j) {
166  auto const &ops = qn_op.at(vertices.second.at(j));
167  qn_sink[j] = ops.at(indices_sink[j]);
168  qn_all[vertices.second.at(j)] = qn_sink[j];
169  }
170 
171  Vector p_si{0, 0, 0};
172  for (auto const &q : qn_sink) {
173  p_si += q.momentum;
174  }
175 
176  // In case that momentum is conserved we add the operators to the
177  // lookup table. Here it is not grouped by source and sink but rather
178  // in that global ordering.
179  if (p_so == -p_si) {
180  quantum_numbers.push_back(qn_all);
181  }
182  }
183  }
184  }

271  std::vector<std::vector<QuantumNumbers>> const &quantum_numbers,
272  std::vector<VdaggerVQuantumNumbers> &vdaggerv_lookup,
273  std::vector<std::vector<std::pair<ssize_t, bool>>> &vdv_indices) {
274  for (auto const &qn_vec : quantum_numbers) {
275  std::vector<std::pair<ssize_t, bool>> vdv_indices_row;
276  for (auto const &qn : qn_vec) {
277  bool dagger;
278  // checking if the combination of quantum numbers already exists in
279  // vdaggerv. The position is stored in the iterator it.
280  auto it = std::find_if(vdaggerv_lookup.begin(),
281  vdaggerv_lookup.end(),
282  [&qn, &dagger](VdaggerVQuantumNumbers vdv_qn) {
283  auto c1 = (vdv_qn.displacement == qn.displacement);
284  auto c2 = (Vector(vdv_qn.momentum.data()) == qn.momentum);
285  // also negative momentum is checked
286  auto c3 =
287  (Vector(vdv_qn.momentum.data()) == (-1) * qn.momentum);
289  Vector const zero(0, 0, 0);
290  if (c1 and c2) {
291  dagger = false;
292  return true;
293  } else if ((c1 and c3) and (qn.displacement.empty())) {
294  dagger = true;
295  return true;
296  } else
297  return false;
298  });
299  // If the quantum number combination already exists only the id is needed
300  // otherwise a new element is created at the end of the lookuptable.
301  if (it != vdaggerv_lookup.end()) {
302  vdv_indices_row.emplace_back((*it).id, dagger);
303  } else {
304  vdaggerv_lookup.emplace_back(
305  VdaggerVQuantumNumbers(ssize(vdaggerv_lookup),
306  {qn.momentum[0], qn.momentum[1], qn.momentum[2]},
307  qn.displacement));
308  vdv_indices_row.emplace_back(vdaggerv_lookup.back().id, false);
309  }
310  }
311  vdv_indices.emplace_back(vdv_indices_row);
312  }

343 static std::vector<std::pair<ssize_t, ssize_t>> create_rnd_vec_id(
344  std::vector<quark> const &quarks,
345  ssize_t const id_q1,
346  ssize_t const id_q2,
347  bool const C1) {
348  // set start and end points of rnd numbers
349  auto rndq1_start = 0;
350  for (auto i = 0; i < id_q1; i++)
351  rndq1_start += quarks[i].number_of_rnd_vec;
352  auto rndq2_start = 0;
353  for (auto i = 0; i < id_q2; i++)
354  rndq2_start += quarks[i].number_of_rnd_vec;
355 
356  auto rndq1_end = rndq1_start + quarks[id_q1].number_of_rnd_vec;
357  auto rndq2_end = rndq2_start + quarks[id_q2].number_of_rnd_vec;
358 
359  // check if there are enough random vectors
360  if ((quarks[id_q1].number_of_rnd_vec < 1) || (quarks[id_q2].number_of_rnd_vec < 1) ||
361  (id_q1 == id_q2 && quarks[id_q1].number_of_rnd_vec < 2)) {
362  std::cerr << "There are not enough random vectors for charged correlators"
363  << std::endl;
364  exit(-1);
365  }
366 
367  // finally filling the array
368  std::vector<std::pair<ssize_t, ssize_t>> rnd_vec_comb;
369  if (!C1) {
370  for (ssize_t i = rndq1_start; i < rndq1_end; ++i)
371  for (ssize_t j = rndq2_start; j < rndq2_end; ++j)
372  // To avoid bias, different random vectors must have different indices.
373  if (i != j) {
374  rnd_vec_comb.emplace_back(i, j);
375  }
376  } else {
377  for (ssize_t i = rndq1_start; i < rndq1_end; ++i)
378  // if C1 == True there is only one random vector and thus only same index
379  // combinations are possible
380  rnd_vec_comb.emplace_back(i, i);
381  }
382 
383  return rnd_vec_comb;
384 }
