419 void build_VdaggerV_lookup(
420  const std::vector<std::vector<QuantumNumbers>> &quantum_numbers,
421  std::vector<VdaggerVQuantumNumbers> &vdaggerv_lookup,
422  std::vector<std::vector<std::pair<ssize_t, bool>>> &vdv_indices) {
423  for (const auto &qn_vec : quantum_numbers) {
424  std::vector<std::pair<ssize_t, bool>> vdv_indices_row;
425  for (const auto &qn : qn_vec) {
426  bool dagger;
427  // checking if the combination of quantum numbers already exists in
428  // vdaggerv. The position is stored in the iterator it.
429  auto it = std::find_if(vdaggerv_lookup.begin(),
430  vdaggerv_lookup.end(),
431  [&qn, &dagger](VdaggerVQuantumNumbers vdv_qn) {
432  auto c1 = (vdv_qn.displacement == qn.displacement);
433  auto c2 = (Vector(vdv_qn.momentum.data()) == qn.momentum);
434  // also negative momentum is checked
435  auto c3 =
436  (Vector(vdv_qn.momentum.data()) == (-1) * qn.momentum);
438  const Vector zero(0, 0, 0);
439  if (c1 and c2) {
440  dagger = false;
441  return true;
442  } else if ((c1 and c3) and (qn.displacement.empty())) {
443  dagger = true;
444  return true;
445  } else
446  return false;
447  });
448  // If the quantum number combination already exists only the id is needed
449  // otherwise a new element is created at the end of the lookuptable.
450  if (it != vdaggerv_lookup.end()) {
451  vdv_indices_row.emplace_back((*it).id, dagger);
452  } else {
453  vdaggerv_lookup.emplace_back(
454  VdaggerVQuantumNumbers(ssize(vdaggerv_lookup),
455  {qn.momentum[0], qn.momentum[1], qn.momentum[2]},
456  qn.displacement));
457  vdv_indices_row.emplace_back(vdaggerv_lookup.back().id, false);
458  }
459  }
460  vdv_indices.emplace_back(vdv_indices_row);
461  }

492 static std::vector<std::pair<ssize_t, ssize_t>> create_rnd_vec_id(
493  const std::vector<quark> &quarks,
494  const ssize_t id_q1,
495  const ssize_t id_q2,
496  const bool C1) {
497  // set start and end points of rnd numbers
498  auto rndq1_start = 0;
499  for (auto i = 0; i < id_q1; i++)
500  rndq1_start += quarks[i].number_of_rnd_vec;
501  auto rndq2_start = 0;
502  for (auto i = 0; i < id_q2; i++)
503  rndq2_start += quarks[i].number_of_rnd_vec;
504 
505  auto rndq1_end = rndq1_start + quarks[id_q1].number_of_rnd_vec;
506  auto rndq2_end = rndq2_start + quarks[id_q2].number_of_rnd_vec;
507 
508  // check if there are enough random vectors
509  if ((quarks[id_q1].number_of_rnd_vec < 1) || (quarks[id_q2].number_of_rnd_vec < 1) ||
510  (id_q1 == id_q2 && quarks[id_q1].number_of_rnd_vec < 2)) {
511  std::cerr << "There are not enough random vectors for charged correlators"
512  << std::endl;
513  exit(-1);
514  }
515 
516  // finally filling the array
517  std::vector<std::pair<ssize_t, ssize_t>> rnd_vec_comb;
518  if (!C1) {
519  for (ssize_t i = rndq1_start; i < rndq1_end; ++i)
520  for (ssize_t j = rndq2_start; j < rndq2_end; ++j)
521  // To avoid bias, different random vectors must have different indices.
522  if (i != j) {
523  rnd_vec_comb.emplace_back(i, j);
524  }
525  } else {
526  for (ssize_t i = rndq1_start; i < rndq1_end; ++i)
527  // if C1 == True there is only one random vector and thus only same index
528  // combinations are possible
529  rnd_vec_comb.emplace_back(i, i);
530  }
531 
532  return rnd_vec_comb;
533 }

841 static void build_general_lookup(
842  std::string const &name,
843  OuterLookup const &ll,
844  std::vector<quark> const &quarks,
845  std::vector<int> const &quark_numbers,
846  int start_config,
847  const std::string &path_output,
848  std::vector<std::vector<QuantumNumbers>> const &quantum_numbers,
849  std::vector<std::vector<std::pair<ssize_t, bool>>> const &vdv_indices) {
850  std::vector<ssize_t> ql_ids(ll.inner.size());
851 
852  // Build the correlator and dataset names for hdf5 output files
853  std::vector<std::string> quark_types;
854  for (const auto &id : quark_numbers)
855  quark_types.emplace_back(quarks[id].type);
856 
857  for (ssize_t d = 0; d < ssize(quantum_numbers); ++d) {
858  for (auto const &lle : ll.inner) {
859  auto const ric_ids = create_rnd_vec_id(quarks,
860  quark_numbers[lle.q1],
861  quark_numbers[lle.q2],
862  name == "C1" || lle.is_q1);
863  build_Quarkline_lookup_one_qn(lle.q2,
864  quantum_numbers[d],
865  vdv_indices[d],
866  ric_ids,
867  *lle.quarkline_lookup,
868  ql_ids);
869  }
870 
871  std::string hdf5_dataset_name = build_hdf5_dataset_name(
872  name, start_config, path_output, quark_types, quantum_numbers[d]);
873 
874  std::vector<ssize_t> ql_ids_new;
875  if (ssize(ll.candidate_factories) == 0) {
876  ql_ids_new = ql_ids;
877  } else {
878  for (auto const &candidate_factory : ll.candidate_factories) {
879  auto const id = candidate_factory->make(ql_ids);
880  ql_ids_new.push_back(id);
881  }
882  }
883 
884  DiagramIndex candidate(ssize(*ll.c_look), hdf5_dataset_name, ql_ids_new);
885 
887  auto it = std::find(ll.c_look->begin(), ll.c_look->end(), candidate);
888 
889  if (it == ll.c_look->end()) {
890  ll.c_look->push_back(candidate);
891  }
892  }
