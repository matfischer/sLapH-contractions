376 template <size_t rvecs>
377 std::map<UpTo, Complex> sub_accumulate(GlobalData const &gd,
378  std::vector<DilutedTrace<rvecs>> const &traces) {
379  // Extract the number of random vectors per quark flavor.
380  auto const &quarks = gd.quarks;
381  UpTo rnd_count;
382  std::transform(std::begin(quarks),
383  std::end(quarks),
384  std::back_inserter(rnd_count),
385  [](quark const &q) { return q.number_of_rnd_vec; });
386 
387  // Obtain the offsets for the random vector ids per quark flavor. This basically is an
388  // exclusive scan, but this is not in the standard library until C++17.
389  UpTo rnd_offset;
390  rnd_offset.push_back(0);
391  std::partial_sum(
392  std::begin(rnd_count), std::end(rnd_count), std::back_inserter(rnd_offset));
393 
394  std::map<UpTo, Complex> results;
395 
396  // We start with the case where in every flavor we only use 1 random vector.
397  UpTo upto(quarks.size(), 1);
398 
399  // We want to go until we have used all available random vectors.
400  UpTo upto_max = rnd_count;
401 
402  // Accumulate the individual contributions into a bucket labeled with the highest random
403  // vector index used per flavor.
404  for (auto const &trace : traces) {
405  auto const &max_used = get_max_used(trace, rnd_offset);
406  results[max_used] += trace.data;
407  }
408 
409  // In the result we want to also accumulate everything below into the current one. There
410  // might be an efficient n-dimensional scan algorithm, but this will very likely not be
411  // the performance critical part, therefore we just implement it in an easy way with a
412  // deep copy.
413  auto const parts = results;
414 
415  for (auto &result : results) {
416  int summands = 0;
417  for (auto const &part : parts) {
418  // We must skip the element itself, otherwise it would be accumulated onto itself.
419  if (result.first == part.first) {
420  continue;
421  }
422 
423  // Take the difference in the maximum random vector index for each flavor.
424  auto diff = result.first;
425  for (int i = 0; i < diff.size(); ++i) {
426  diff[i] -= part.first[i];
427  }
428 
429  // Are none the differences positive? This means that all random vector indices used
430  // are smaller or equal and therefore this contributed. The case where all
431  // differences are zero is excluded already at this point.
432  bool const none_positive =
433  std::none_of(std::begin(diff), std::end(diff), [](RndId d) { return d > 0; });
434  // If so, accumulate this onto the result.
435  if (none_positive) {
436  result.second += part.second;
437  ++summands;
438  }
439  }
440 
441  // Normalize.
442  result.second /= summands;
443  }
444 
445  return results;
446 }
