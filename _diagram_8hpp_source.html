117  void assemble(int const t, BlockIterator const &slice_pair, DiagramParts &q) override {
118  int const tid = omp_get_thread_num();
119 
120  for (int i = 0; i != ssize(corr_lookup()); ++i) {
121  c_[tid][i] = Numeric{};
122  }
123 
124  assemble_impl(c_.at(tid), slice_pair, q);
125 
126  {
127  std::lock_guard<std::mutex> lock(mutexes_[t]);
128 
129  for (int i = 0; i != ssize(corr_lookup()); ++i) {
130  correlator_[t][i] += c_[tid][i];
131  }
132  }
133  }
134 
135  void write() override {
136  assert(output_path_ != "");
137  assert(output_filename_ != "");
138 
139  WriteHDF5Correlator filehandle(
140  output_path_, name(), output_filename_, comp_type_factory<Numeric>());
141 
142  std::vector<Numeric> one_corr(Lt_);
143 
144  for (int i = 0; i != ssize(corr_lookup()); ++i) {
145  for (int t = 0; t < Lt_; ++t) {
146  one_corr[t] = correlator_[t][i] / static_cast<double>(Lt_);
147  }
148  // Write data to file.
149  filehandle.write(one_corr, corr_lookup()[i]);
150  }
151  }
