229 Operators make_operator_list(const std::string &operator_string) {
230  Operators op_list; // return object
231 
232  // Two steps are necessary:
233  // 1. Getting all operators in one list which are separated by ";"
234  // 2. Separating the individual operators into its smaller bits, which are
235  // separated by "."
236  // Tokenize the string on the ";" delimiter -> Individual operators
237  std::vector<std::string> operator_tokens;
238  boost::split(operator_tokens, operator_string, boost::is_any_of(":"));
239 
240  // running over opeator tokens and split them further (Step 2):
241  for (const auto &op_t : operator_tokens) {
242  std::vector<std::string> tokens;
243  boost::split(tokens, op_t, boost::is_any_of("."));
244  std::vector<int> gammas;
245  std::vector<DisplacementDirection> disp_dirs;
246  std::vector<std::vector<Vector>> mom_vec;
247  for (auto& str : tokens) {
248  // getting the gamma structure
249  if (str.compare(0, 1, "g") == 0)
250  gammas.push_back(boost::lexical_cast<int>(str.erase(0, 1)));
251  // getting the displacement indices
252  else if (str.compare(0, 1, "d") == 0) {
253  // 0 encodes no displacement. disp_dir remains empty in this case
254  if (str.compare(1, 1, "0") == 0) {
255  disp_dirs.push_back( DisplacementDirection() );
256  }
257  else if ((str.compare(1, 1, "<") == 0) || (str.compare(1, 1, ">") == 0))
258  create_displacement_direction_from_string(str, disp_dirs);
259  else {
260  std::cout << "Something wrong with the displacement in the operator"
261  " definition"
262  << std::endl;
263  exit(0);
264  }
265  }
266  // getting the momenta
267  else if (str.compare(0, 1, "p") == 0) {
268  if (str.compare(1, 1, "(") == 0) {
269  mom_vec.resize(1);
270  mom_vec[0].push_back(create_3darray_from_string(str));
271  }
272  else {
273  create_mom_array_from_string(str, mom_vec);
274  }
275  }
276  // catching wrong entries
277  else {
278  std::cout << "there is something wrong with the operators" << std::endl;
279  exit(0);
280  }
281  }
282 
283  for (const auto &mom_vec_tmp : mom_vec) { // momenta
284  for (auto mom : mom_vec_tmp) {
285  for (auto &disp_dir : disp_dirs) {
286  op_list.push_back({gammas, disp_dir, mom});
287  }
288  }
289  }
290  }
