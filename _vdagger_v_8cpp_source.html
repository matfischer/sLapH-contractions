95 // creates a two-dimensional vector containing the momenta for the operators
96 // input: Lx, Ly, Lz -> lattice extend in x, y, and z direction

99 void create_momenta(const ssize_t Lx, const ssize_t Ly, const ssize_t Lz,
100  const std::vector<VdaggerVQuantumNumbers>& vdaggerv_lookup,
101  array_cd_d2& momentum){
102 
103  static const std::complex<double> I(0.0, 1.0);
104 
105  // To calculate Vdagger exp(i*p*x) V only the momenta corresponding to the
106  // quantum number id in op_VdaggerV will be used. The rest can be obtained
107  // by adjoining
108  for(const auto& op : vdaggerv_lookup){
109  // op_VdaggerV contains the index of one (redundancy) op_Corr which
110  // allows to deduce the quantum numbers (momentum)
111  const double ipx = op.momentum[0] * 2. * M_PI / (double) Lx;
112  const double ipy = op.momentum[1] * 2. * M_PI / (double) Ly;
113  const double ipz = op.momentum[2] * 2. * M_PI / (double) Lz;
114  // calculate \vec{p} \cdot \vec{x} for all \vec{x} on the lattice
115  for(int x = 0; x < Lx; ++x){
116  const int xH = x * Ly * Lz; // helper variable
117  const double ipxH = ipx * x; // helper variable
118  for(int y = 0; y < Ly; ++y){
119  const int xHyH = xH + y * Lz; // helper variable
120  const double ipxHipyH = ipxH + ipy * y; // helper variable
121  for(int z = 0; z < Lz; ++z){
122  // multiply \vec{p} \cdot \vec{x} with complex unit and exponentiate
123  momentum[op.id][xHyH + z] = exp(-I * (ipxHipyH + ipz * z));
124  }}}//loops over spatial vectors end here
125  }//loop over redundant quantum numbers ends here
126 }

129 void build_and_write_vdaggerv(const ssize_t Lt, const ssize_t Lx, const ssize_t Ly,
130  const ssize_t Lz, const ssize_t nb_ev,
131  const OperatorLookup& operator_lookuptable,
132  const std::string& filename_ev,
133  const std::string& pathname_vdaggerv,
134  const std::string& filename_vdaggerv){
135 
136  clock_t t2 = clock();
137  const ssize_t dim_row = 3*Lx*Ly*Lz;
138  const ssize_t id_unity = operator_lookuptable.index_of_unity;
139 
140  array_cd_d2 momentum;
141  momentum.resize(boost::extents[
142  operator_lookuptable.vdaggerv_lookup.size()][Lx*Ly*Lz]);
143  create_momenta(Lx, Ly, Lz, operator_lookuptable.vdaggerv_lookup, momentum);
144 
145 #pragma omp parallel
146 {
147  Eigen::VectorXcd mom = Eigen::VectorXcd::Zero(dim_row);
148  Eigen::MatrixXcd vdaggerv = Eigen::MatrixXcd::Zero(nb_ev, nb_ev);
149  Eigen::MatrixXcd V_t = Eigen::MatrixXcd::Zero(dim_row, nb_ev);
150 
151  #pragma omp for schedule(dynamic)
152  for(ssize_t t = 0; t < Lt; ++t){
153 
154  // creating full filename for eigenvectors and reading them in
155  char infile[200];
156  sprintf(infile, "%s.%03d", filename_ev.c_str(), (int) t);
157  read_eigen_vector(infile, 0, V_t); // reading eigenvectors
158 
159  // VdaggerV is independent of the gamma structure and momenta connected by
160  // sign flip are related by adjoining VdaggerV. Thus the expensive
161  // calculation must only be performed for a subset of quantum numbers given
162  // in op_VdaggerV.
163  for(const auto& op : operator_lookuptable.vdaggerv_lookup){
164  // For zero momentum and displacement VdaggerV is the unit matrix, thus
165  // the calculation is not performed
166  if(op.id != id_unity){
167  // momentum vector contains exp(-i p x). Divisor 3 for colour index.
168  // All three colours on same lattice site get the same momentum.
169  for(ssize_t x = 0; x < dim_row; ++x) {
170  mom(x) = momentum[op.id][x/3];
171  }
172  vdaggerv = V_t.adjoint() * mom.asDiagonal() * V_t;
173 
174  // write operators to file
175  std::string dummy = filename_vdaggerv + ".p_" +
176  std::to_string(op.momentum[0]) +
177  std::to_string(op.momentum[1]) +
178  std::to_string(op.momentum[2]);
179  char outfile[200];
180  sprintf(outfile, "%s_.t_%03d", dummy.c_str(), (int) t);
181  write_vdaggerv(pathname_vdaggerv, std::string(outfile), vdaggerv);
182  }
183  }
184  } // loop over time
185 }// pragma omp parallel ends here
186 
187  t2 = clock() - t2;
188  std::cout << std::setprecision(1) << "\t\t\tSUCCESS - " << std::fixed
189  << ((float) t2)/CLOCKS_PER_SEC << " seconds" << std::endl;
190 
191 }
