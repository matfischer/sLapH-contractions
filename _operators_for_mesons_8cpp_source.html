21 void create_momenta(const size_t Lx,
22  const size_t Ly,
23  const size_t Lz,
24  const std::vector<VdaggerVQuantumNumbers> &vdaggerv_lookup,
25  array_cd_d2 &momentum) {
26  static const std::complex<double> I(0.0, 1.0);
27 
32  for (const auto &op : vdaggerv_lookup) {
33  // op_VdaggerV contains the index of one (redundancy) op_Corr which
34  // allows to deduce the quantum numbers (momentum)
35  const double ipx = op.momentum[0] * 2. * M_PI / (double)Lx;
36  const double ipy = op.momentum[1] * 2. * M_PI / (double)Ly;
37  const double ipz = op.momentum[2] * 2. * M_PI / (double)Lz;
38  // calculate \vec{p} \cdot \vec{x} for all \vec{x} on the lattice
39  for (int x = 0; x < Lx; ++x) {
40  const int xH = x * Ly * Lz; // helper variable
41  const double ipxH = ipx * x; // helper variable
42  for (int y = 0; y < Ly; ++y) {
43  const int xHyH = xH + y * Lz; // helper variable
44  const double ipxHipyH = ipxH + ipy * y; // helper variable
45  for (int z = 0; z < Lz; ++z) {
46  // multiply \vec{p} \cdot \vec{x} with complex unit and exponentiate
47  momentum[op.id][xHyH + z] = exp(-I * (ipxHipyH + ipz * z));
48  }
49  }
50  } // loops over spatial vectors end here
51  } // loop over redundant quantum numbers ends here
52 }

132 void OperatorFactory::build_vdaggerv(const std::string &filename, const int config) {

140  const std::string full_path(dummy_path);

143  std::cout << "\tdirectory " << full_path.c_str()

170  if(need_gauge) gauge.read_gauge_field(config,0,Lt-1);

172 #pragma omp for schedule(dynamic)

175  if (!(operator_lookuptable.vdaggerv_lookup.size() == 1 &&
176  operator_lookuptable.vdaggerv_lookup[0].id == id_unity)) {

183  // sign flip are related by adjoining VdaggerV. Thus the expensive
184  // calculation must only be performed for a subset of quantum numbers given
185  // in op_VdaggerV.
186  for (const auto &op : operator_lookuptable.vdaggerv_lookup) {
187  // For zero momentum and displacement VdaggerV is the unit matrix, thus
188  // the calculation is not performed

191  Eigen::MatrixXcd W_t;

195  }
196  else {
197  W_t = V_t[0];
198  }
199  // momentum vector contains exp(-i p x). Divisor 3 for colour index.
200  // All three colours on same lattice site get the same momentum.
201  for (size_t x = 0; x < dim_row; ++x) {
202  mom(x) = momentum[op.id][x / 3];
203  }
204  vdaggerv[op.id][t] = V_t[0].adjoint() * mom.asDiagonal() * W_t;
205  // writing vdaggerv to disk
206  if (handling_vdaggerv == "write") {
207  char dummy2[200];
208  sprintf(dummy2, "operators.%04d.p_", config);
209  std::string dummy = std::string(dummy2) + std::to_string(op.momentum[0]) +
210  std::to_string(op.momentum[1]) +
211  std::to_string(op.momentum[2]);
212  char outfile[200];
213  sprintf(outfile, "%s_.t_%03d", dummy.c_str(), (int)t);
214  write_vdaggerv(full_path, std::string(outfile), vdaggerv[op.id][t]);
215  }
216  } else // zero momentum
217  vdaggerv[op.id][t] = Eigen::MatrixXcd::Identity(nb_ev, nb_ev);
218  }
219  } // loop over time
220  } // pragma omp parallel ends here
221
